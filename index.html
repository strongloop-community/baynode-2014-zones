var http = require('http');
var cache = {};

function curl(url, cb) {
  if (cache[url])
    return cb(null, cache[url]);

  var data = '';

  http.get(url, function(res) {
    res.setEncoding('utf8');

    res.on('data', function(s) {
      data += s;
    });

    res.on('end', function() {
      cache[url] = data;
      cb(null, data);
    });

    res.on('error', function(err) {
      cb(err);
    });
  });
}
